import java.util.Scanner;

class Change{
    int dollar_count;
    int quarter_count;
    int dime_count;
    int nickel_count;
    double total;
}

public class ChangeMachine {
    public static void main(String[] args) {
        Scanner user_input = new Scanner(System.in);
        Change myChange = new Change();

        System.out.print("Enter whole dollars: ");
        myChange.dollar_count = user_input.nextInt();

        System.out.print("Enter number of quarters: ");
        myChange.quarter_count = user_input.nextInt();

        System.out.print("Enter number of dimes: ");
        myChange.dime_count = user_input.nextInt();

        System.out.print("Enter number of nickels: ");
        myChange.nickel_count = user_input.nextInt();

        myChange.total = myChange.dollar_count + 0.25 * myChange.quarter_count
                                                   + 0.1 * myChange.dime_count
                                                   + 0.05 * myChange.nickel_count;
        System.out.println("You have: $" + myChange.total);
    }
