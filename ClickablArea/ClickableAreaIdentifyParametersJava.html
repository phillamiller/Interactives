Knowledge Check: Identify Parameters

Click on all parameters in the following code.

Remember that parameters are variables.


class Main {
  public static double get_diameter(double radius) {
    return radius * 2;
  }
 
  public static double get_area(double radius) {
    double area = pi * Math.pow(radius, 2);

    return area;
  }

  public static double get_circumference(double radius) {
    double diameter = get_diameter(radius);
    double circumference = pi * diameter;

    return circumference;
  }
  
  public static void display_area_and_circumference(double area, double circumference) {
    System.out.println(area + ", " + circumference);
  }

  static double pi = 3.1415;

  public static void main(String[] args) {
    double radius = 100;
    double area = get_area(radius);
    double circumference = get_circumference(radius);

    display_area_and_circumference(area, circumference);
  }
}
