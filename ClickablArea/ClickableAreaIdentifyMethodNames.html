Click on all method names in the code below - wether the name is part of a method definition or a method call.


import java.util.Scanner;
public class Dog {
    int age;
    String name;
    String breed;
    String color;

    void bark() {
        System.out.println("Bark! Bark! says the " + color + " dog.");
    }

    void hungry(){
        System.out.println(name + " is whining for dinner.");
    }

    void sleep() {
        System.out.println(breed + "s are always so cute when they're sleeping.");
    }
}

public class DogsAreAwesome {
    public static void main(String []args) {
        Scanner user_input = new Scanner(System.in);
        Dog myDog = new Dog();

        System.out.print("What's your dog's name? ");
        myDog.name = user_input.next();

        System.out.println(myDog.name + " is a great name for a dog!");
        myDog.hungry();

        System.out.print("How old are they? ");
        myDog.age = user_input.nextInt();

        System.out.print("What breed is your dog? ");
        myDog.breed = user_input.next();
        myDog.sleep();

        System.out.print("What color is your dog? ");
        myDog.color = user_input.next();
        myDog.bark();
    }
}
